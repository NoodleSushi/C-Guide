enum enum_name
{
    const1, // 0
    const2, // 1
    const3, // 2
    const4, // 3
} optional_enum_var;

typedef enum optional_enum_name
{
    const1, // 0
    const2, // 1
    const3, // 2
    const4, // 3
} enum_name;

enum optional_enum_name my_enum1;
enum_name my_enum2;

enum my_enum
{
                  //value| increment
    const_01,     //  0  |  0 + 0 ⬎
    const_02,     //  1  |  0 + 1 ↓
    const_03,     //  2  |  0 + 2 ↓
    const_04,     //  3  |  0 + 3 ↓
    const_05 = 5, //  5  |  5 + 0 ⬎
    const_06,     //  6  |  5 + 1 ↓
    const_07,     //  7  |  5 + 2 ↓
    const_08,     //  8  |  5 + 3 ↓
    const_09 = 1, //  1  |  1 + 0
    const_10 = 3, //  3  |  3 + 0
    const_11 = 6, //  6  |  6 + 0 ⬎
    const_12,     //  7  |  6 + 1 ↓
    const_13,     //  8  |  6 + 2 ↓
    const_14,     //  9  |  6 + 3 ↓
};
